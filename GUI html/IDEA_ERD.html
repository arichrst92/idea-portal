-- ── SP THRESHOLD CHECK FUNCTION ────────────────────────────────────
CREATE OR REPLACE FUNCTION check_sp_threshold() RETURNS TRIGGER AS $$
BEGIN
  -- Check if employee scored < 60 for 3 consecutive months
  IF (SELECT COUNT(*) FROM performance_assessments
      WHERE employee_id = NEW.employee_id
        AND final_score < 60
        AND status = 'submitted'
        AND (period_year, period_month) >= (
            SELECT period_year, period_month FROM performance_assessments
            ORDER BY period_year DESC, period_month DESC LIMIT 1 OFFSET 2
        )) >= 3
  THEN
    -- Create SP trigger notification to Operation
    INSERT INTO notifications(recipient_id, type, title, entity_type, entity_id)
    SELECT u.id, 'alert', 'SP Threshold Triggered for ' || e.full_name,
           'employee', NEW.employee_id
    FROM employees e JOIN users u ON u.employee_id = e.id
    WHERE e.department_id = (SELECT department_id FROM departments WHERE code = 'OPS');
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_sp_threshold
AFTER INSERT OR UPDATE ON performance_assessments
FOR EACH ROW EXECUTE FUNCTION check_sp_threshold();
